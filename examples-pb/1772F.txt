struct picture_line {
    string<n> fake int32[n] mapper m line;
    eoln;
}

struct picture {
    eoln;
    picture_line[n] lines; 
}

struct input {
    int32 n;
    int32 m;
    int32 k;
    eoln;
    picture[k + 1] pictures;
}

enum operation_type mapped by int32 {
    PAINT = 1;
    COPY = 2;
}

struct operation {
    operation_type op;
    int32 x if op == PAINT;
    int32 y if op == PAINT;
    int32 i if op == COPY;
    eoln;
}

output {
    int32 start_pic_index;
    eoln;
    int32 q;
    eoln;
    operation[q] operations;
}
